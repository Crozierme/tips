# 將多個 plot 畫在一個 figure


# 指定 figure 的編號

plt.figure(1) 

# subplot(numrows, numcols, fignum)
# numrows, numcols 指定 figure 裡有幾個 plot
# fignum < numrows * numcols 為 plot 的編號

# 若 numrows * numcols < 10
# 則可以寫成三個數字連在一起
# 譬如 subplot(2,1,2) 跟 subplot(212) 效果相同

plt.subplot(2,1,1) 
plt.plot(data1, data2)

plt.subplot(212) 
plt.plot(data3, data4)

plt.show()
In [9]:
plt.figure(1)                # the first figure
plt.subplot(2,1,1)           # the first subplot in the first figure
plt.plot([1, 2, 3])
plt.subplot(2,1,2)           # the second subplot in the first figure
plt.plot([4, 5, 6])

plt.figure(2)                # a second figure
plt.plot([4, 5, 6])          # creates a subplot(111) by default

plt.figure(1)                # figure 1 current; subplot(212) still current
plt.subplot(2,1,1)           # make subplot(211) in figure1 current
plt.title('Easy as 1, 2, 3') # subplot 211 title
plt.show()
