Metropolis-Hastings 演算法¶

  • : $\theta_t$
  • : $\theta_p\sim N(\theta_t, \sigma)$
  • $\rho$ $$\rho=\frac{\Pi(s_p|D)}{\Pi(s_t|D)}=\frac{f(D|s_p)\pi(s_p)}{f(D|s_t)\pi(s_t)}$$
  • $\theta_{t+1}$
    • $\rho \geq 1$, $\theta_{t+1} = \theta_p$
    • $\rho < 1$, $u\sim U(0, 1)$, $u\leq \rho$, $\theta_{t+1} = \theta_p$, $\theta_{t+1}=\theta_t$
  • $t$ , $\theta$

def target(lik, prior, n, h, theta):
    if theta < 0 or theta > 1:
        return 0
    else:
        return lik(n, theta).pmf(h)*prior.pdf(theta)

#  Markov chain
def gen_markov_chain(niters, n, h, theta, lik, prior, sigma):
    samples = np.zeros(niters+1)
    samples[0] = theta
    for i in range(niters):
        theta_p = st.norm(theta, sigma).rvs()
        rho = min(1, target(lik, prior, n, h, theta_p)/target(lik, prior, n, h, theta))
        u = np.random.uniform()
        if u <= rho:
            #naccept = naccept+1
            theta = theta_p
        samples[i+1] = theta
    return samples
